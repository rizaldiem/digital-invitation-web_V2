<script>
let loadComments; // Dideklarasikan di global
let lastId = 0;   // Dideklarasikan di global

document.addEventListener('DOMContentLoaded', function () {
    const container = document.getElementById('realtime-comments');
    if (!container) return;

    const postId = container.getAttribute('data-post-id');

    loadComments = async function (page = 1) {
    try {
        const list = container.querySelector('ul'); // target ul di dalam container
        if (list) list.classList.add('fade-out');   // mulai fade-out

        const response = await fetch('https://attarivitation.com/wp-admin/admin-ajax.php?action=get_realtime_comments&post_id=' + postId + '&page=' + page);
        const result = await response.json();

        if (result.success && result.data.updated) {
            setTimeout(() => {
                container.innerHTML = result.data.html; // replace isi container (ul ikut ke-replace)

                // reset scroll ke atas
                container.scrollTop = 0;

                // ambil ul baru setelah replace
                const newList = container.querySelector('ul');
                if (newList) newList.classList.remove('fade-out'); // fade-in lagi
            }, 300); // durasi sama dengan CSS transition
        }
    } catch (error) {
        console.error('Error loading comments:', error);
        const list = container.querySelector('ul');
        if (list) list.classList.remove('fade-out'); // fallback
    }
};



    // 🔃 Load pertama
    loadComments();

    // ✅ Event hapus komentar
    container.addEventListener('click', function(e) {
        if (e.target.classList.contains('delete-comment-btn')) {
            const commentId = e.target.getAttribute('data-comment-id');
            if (confirm('Yakin mau hapus komentar ini?')) {
                fetch('https://attarivitation.com/wp-admin/admin-ajax.php', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                    body: new URLSearchParams({
                        action: 'delete_comment',
                        comment_id: commentId,
                        _ajax_nonce: 'eddd6a2a48'
                    })
                })
                .then(res => res.json())
                .then(data => {
                    if (data.success) {
                        loadComments();
                    } else {
                        alert('Gagal hapus komentar: ' + (data.data || 'Unknown error'));
                    }
                })
                .catch(error => console.error('Error deleting comment:', error));
            }
        }
    });

    // ✅ Event pagination (angka / prev / next)
    container.addEventListener('click', function(e) {
        if (e.target.classList.contains('page-btn')) {
            const newPage = parseInt(e.target.getAttribute('data-page'));
            loadComments(newPage); // load page baru
        }
    });
});

<script>
jQuery(document).ready(function($) {
    const targetFormId = 'formRSVP'; // ID form target

    function refreshComments(source = '') {
        console.log(`🔄 Refresh komentar via: ${source}`);
        if (typeof loadComments === 'function') {
            lastId = 0;
            loadComments();
        } else {
            console.warn('⚠️ loadComments tidak ditemukan');
        }
    }

    // ✅ Event sukses submit
    $(document).on('elementor-pro/forms/submit/success', function(event, response, form) {
        if (form.attr('id') === targetFormId) {
            refreshComments('event sukses');
        }
    });

    // ✅ Event gagal submit
    $(document).on('elementor-pro/forms/submit/error', function(event, response, form) {
        if (form.attr('id') === targetFormId) {
            refreshComments('event gagal');
        }
    });

    // ✅ Fallback: MutationObserver untuk ngawasin perubahan pesan form
    const targetForm = document.getElementById(targetFormId);
    if (targetForm) {
        const observer = new MutationObserver(mutations => {
            mutations.forEach(mutation => {
                mutation.addedNodes.forEach(node => {
                    if (node.nodeType === 1) {
                        // Kalau ada pesan sukses atau error nambah di form
                        if (node.classList.contains('elementor-message') || 
                            node.classList.contains('elementor-message-success') || 
                            node.classList.contains('elementor-message-danger')) {
                            refreshComments('mutation observer');
                        }
                    }
                });
            });
        });

        observer.observe(targetForm, {
            childList: true,
            subtree: true
        });
    }
});

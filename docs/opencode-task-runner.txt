============================== TASK 7 — ADD GUEST PERSONALIZATION
==============================

OPEN CODE PROMPT

1.  Update file /src/app/invite/[slug]/page.tsx

import { notFound } from “next/navigation” import InvitationPage from
“@/invitation/InvitationPage”

export default async function InvitePage({ params, searchParams }: {
params: { slug: string } searchParams: { to?: string } }) {

const slug = params.slug const guest = searchParams.to

if (!slug) { notFound() }

return (
        {guest && (
          <h2 className="text-xl">
            Dear {guest.replace("-", " ")}
          </h2>
        )}

        <h1 className="text-3xl font-bold">
          Invitation for {slug}
        </h1>

      </div>
    </InvitationPage>

) }

End of instructions.
